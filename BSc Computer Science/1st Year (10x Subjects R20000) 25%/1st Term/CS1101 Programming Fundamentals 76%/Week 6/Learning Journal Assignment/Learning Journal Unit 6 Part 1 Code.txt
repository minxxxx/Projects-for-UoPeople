def part1():
    longstring = "blue red green yellow black purple white grey pink orange"
    splitstring = longstring.split()
    print("my string split into a list:")
    print(splitstring)
    splitstring.remove("blue")
    print('')
    print("my string's first deletion of blue:")
    print(splitstring)
    splitstring.pop(0)
    print('')
    print("my string's second deletion of red:")
    print(splitstring)
    del splitstring[0]
    print('')
    print("my string's third deletion of green:")
    print(splitstring)
    splitstring.sort()
    print('')
    print("my list being sorted:")
    print(splitstring)
    splitstring.append("silver")
    print('')
    print("adding the first word silver to my list:")
    print(splitstring)
    splitstring.insert(0, "lilac")
    print('')
    print("adding the second word lilac to my list:")
    print(splitstring)
    splitstring.extend(['gold'])
    print('')
    print("adding a third word gold to my list:")
    print(splitstring)
    longstring = ' '.join(splitstring)
    print('')
    print("my list being joined together into a string again and being printed:")
    print(longstring)
